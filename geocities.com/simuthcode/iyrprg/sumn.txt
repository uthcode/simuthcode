SET TALK OFF
STORE 0 TO N,I,SUM,OSUM
@ 5,5 SAY "ENTER THE NO. UP TO WHICH WE HAVE TO SUM " GET N
READ
CLEAR
@ 6,20 TO 10,50 DOUBLE
TSUM=N*(N+1)/2
@ 7,25 SAY "TSUM="+STR(TSUM)
DO WHILE I<=N
SUM = SUM+I
I=I+2
ENDDO
@8,25 SAY "EVENSUM="+STR(SUM)
OSUM=0
I=1
DO WHILE I<=N
   OSUM=OSUM+I
   I=I+2
ENDDO
@ 9,25 SAY "ODSUM=" + STR(OSUM)
RETURN
